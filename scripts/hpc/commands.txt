# Next sequential jobs after the first one starts or finishes.
SAT_CPUS_PER_TASK=192 bash scripts/hpc/zaratan_spy_saturation.sh
SAT_CPUS_PER_TASK=64 bash scripts/hpc/zaratan_spy_saturation.sh
SAT_RUNNER_OVERSUBSCRIPTION_FACTOR=24 bash scripts/hpc/zaratan_spy_saturation.sh

# If queue time is still bad, try a 1-GPU quick experiment.
SAT_GRES=gpu:a100:1 bash scripts/hpc/zaratan_spy_saturation.sh

# Override the quick walltime explicitly if needed.
SAT_TIME=00:20:00 bash scripts/hpc/zaratan_spy_saturation.sh
SAT_TIME=00:20:00 SAT_GRES=gpu:a100:1 bash scripts/hpc/zaratan_spy_saturation.sh

# Custom multi-job sweep once the queue situation is acceptable.
BUNDLE="$(ls -1t local-training/hpc/bundles/rl-runtime-*.tar.gz | head -1)"
python3 scripts/hpc/spy_saturation.py submit \
  --bundle "$BUNDLE" \
  --tag a100-saturation \
  --partition gpu-a100 \
  --gres gpu:a100:2 \
  --mem 128G \
  --time 02:00:00 \
  --train-profiles 4 \
  --cpus-per-task 64,128,192 \
  --runner-oversubscription-factor 20,24

# Wider CPU sweep if you get a high-core single node and can tolerate queue time.
python3 scripts/hpc/spy_saturation.py submit \
  --bundle "$BUNDLE" \
  --tag high-core \
  --partition gpu-a100 \
  --gres gpu:a100:2 \
  --mem 192G \
  --time 02:00:00 \
  --train-profiles 4 \
  --cpus-per-task 128,192,256,320 \
  --runner-oversubscription-factor 20,24,28
