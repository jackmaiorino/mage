BUNDLE="$(ls -1t local-training/hpc/bundles/rl-runtime-*.tar.gz | head -1)"

  --partition=gpu \
  --gres=gpu:a100:4 \
  --cpus-per-task=96 \
  --mem=192G \
  --time=04:00:00 \
  --job-name=spy-pbt-12x4g \
  --export=ALL,HPC_NATIVE_ORCH=1,PY_SERVICE_MODE=shared_gpu,GPU_SERVICE_PORT_BASE=26100,GPU_SERVICE_METRICS_PORT_BASE=27100,GPU_SERVICE_STARTUP_TIMEOUT_SECONDS=120,MAGE_RL_RUNTIME_TARBALL="$BUNDLE",METRICS_PORT_BASE=19100,TRAIN_PROFILES=8,CPU_HEADROOM=0,RUNNER_OVERSUBSCRIPTION_FACTOR=6,TRAINER_START_STAGGER_SECONDS=20,PY_BRIDGE_CONNECT_RETRIES=60,PY_BRIDGE_CONNECT_RETRY_DELAY_MS=2000,TOTAL_EPISODES=1000000,INFER_WORKERS=1 \
  scripts/hpc/submit_spy_pbt.slurm)
