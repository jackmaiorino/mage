# Single-node saturation sweep with the newest bundle.
bash scripts/hpc/zaratan_spy_saturation.sh

# Custom saturation sweep.
BUNDLE="$(ls -1t local-training/hpc/bundles/rl-runtime-*.tar.gz | head -1)"
python3 scripts/hpc/spy_saturation.py submit \
  --bundle "$BUNDLE" \
  --tag a100-saturation \
  --partition gpu-a100 \
  --gres gpu:a100:2 \
  --mem 128G \
  --time 02:00:00 \
  --train-profiles 2,4 \
  --cpus-per-task 64,128,192 \
  --runner-oversubscription-factor 8,12

# Wider CPU sweep if you get a high-core single node.
python3 scripts/hpc/spy_saturation.py submit \
  --bundle "$BUNDLE" \
  --tag high-core \
  --partition gpu-a100 \
  --gres gpu:a100:2 \
  --mem 192G \
  --time 02:00:00 \
  --train-profiles 4 \
  --cpus-per-task 128,192,256,320 \
  --runner-oversubscription-factor 8,12,16

# Summarize a finished or running sweep from its manifest.
python3 scripts/hpc/spy_saturation.py summarize \
  --manifest local-training/hpc/saturation_runs/<timestamp>_<tag>/manifest.json

# Summarize specific jobs without a manifest.
python3 scripts/hpc/spy_saturation.py summarize --job-id 123456 --job-id 123457
